      margin-top: 1em;

<body>

  <h2>收據圖片上傳（n8n Webhook）</h2>


  <div id="drop-area">

    📷 點擊或拖曳圖片上傳

    <input type="file" id="fileElem" accept="image/*" style="display:none" />

  </div>


  <button id="uploadBtn">🚀 上傳辨識</button>

  <div id="result"></div>


  <script>

    const token = "Boos58609225**";

    const webhookUrl = "https://guruguru.app.n8n.cloud/webhook-test/upload-receipt";


    const dropArea = document.getElementById("drop-area");

    const fileElem = document.getElementById("fileElem");

    const uploadBtn = document.getElementById("uploadBtn");

    let selectedFile = null;


    dropArea.addEventListener("click", () => fileElem.click());

    fileElem.addEventListener("change", handleFiles);


    function handleFiles() {

      selectedFile = this.files[0];

      dropArea.innerText = `✅ 已選擇檔案：${selectedFile.name}`;

    }


    uploadBtn.addEventListener("click", () => {

      if (!selectedFile) return alert("請先選擇圖片！");

      const reader = new FileReader();

      reader.onload = () => {

        const imageBase64 = reader.result;

        fetch(webhookUrl, {

          method: "POST",

          headers: {

            "Content-Type": "application/json"

          },

          body: JSON.stringify({

            token,

            imageBase64

          })

        })

        .then(res => res.json())

        .then(data => {

          document.getElementById("result").innerText =

            data.status === "success"

              ? "✅ 成功辨識並寫入 Google Sheets"

              : "❌ 失敗：" + data.message;

        })

        .catch(err => {

          document.getElementById("result").innerText = "❌ 錯誤：" + err.message;

        });

      };

      reader.readAsDataURL(selectedFile);

    });

  </script>

</body>
